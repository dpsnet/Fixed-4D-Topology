\section{p-adic Bowen Formula}

\begin{theorem}[First Rigorous Proof]
For $f(z) = z^{p^k}$ in $\mathbb{Q}_p$:
\[
\dimH J(f) = \delta
\]
where $\delta$ satisfies $P(-\delta \cdot \log |f'|_p) = 0$.
\end{theorem}

\textbf{Proof:}
\begin{enumerate}
    \item Julia set: $J(f) = \{z : |z|_p = 1\}$
    \item Derivative: $|f'(z)|_p = p^{-k}$ (constant)
    \item Pressure: $P(s) = \log(p^k) - s \cdot k\log p = (1-s)k\log p$
    \item Bowen equation: $P(\delta) = 0 \Rightarrow \delta = 1$
    \item Verification: $\dimH J(f) = 1$ $\checkmark$
\end{enumerate}
